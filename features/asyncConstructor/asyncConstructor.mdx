### asyncConstructor

[source code](_asyncConstructor.ts)

```ts
import asyncConstructor from 'sky/features/asyncConstructor'

class Foo {
    constructor() {
        return asyncConstructor(async () => {
            await some()
            return this
        })
    }
}

const foo = await new Foo()

class Boo extends Foo {
    constructor() {
        const promise = super()

        return asyncConstructor(async () => {
            const _this = await promise
            await some()
            return this
        })
    }
}

const boo = await new Boo()
```
